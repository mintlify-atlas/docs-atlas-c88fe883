---
title: "Export Compliance Report"
api: "GET /api/export"
description: "Export a full compliance report as JSON including scan metadata, violations, rules, and scores"
---

Exports a comprehensive compliance report for a specific scan or the latest scan. The report includes complete scan metadata, all violations with details, review history, policy information, and summary statistics.

## Query Parameters

<ParamField query="scan_id" type="string">
  The unique identifier of the scan to export. If not provided, exports the latest scan for the authenticated user.
</ParamField>

<ParamField query="format" type="string" default="json">
  Export format. Currently only `json` is supported.
</ParamField>

## Response

<ResponseField name="report" type="object">
  The complete compliance report object
  
  <ResponseField name="generated_at" type="string">
    ISO 8601 timestamp of when the report was generated
  </ResponseField>
  
  <ResponseField name="policy" type="object">
    Policy information associated with the scan
    <ResponseField name="id" type="string">
      Policy unique identifier
    </ResponseField>
    <ResponseField name="name" type="string">
      Policy name
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="scan" type="object">
    Scan metadata
    <ResponseField name="id" type="string">
      Scan unique identifier
    </ResponseField>
    <ResponseField name="score" type="number">
      Compliance score (0-100)
    </ResponseField>
    <ResponseField name="violation_count" type="number">
      Total number of violations detected
    </ResponseField>
    <ResponseField name="scan_date" type="string">
      ISO 8601 timestamp of when the scan was created
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="violations" type="array">
    Array of all violation objects with complete details including severity, rule_id, record_id, status, and review information
  </ResponseField>
  
  <ResponseField name="reviews" type="array">
    Array of review history objects
    <ResponseField name="violation_id" type="string">
      ID of the reviewed violation
    </ResponseField>
    <ResponseField name="status" type="string">
      Review status (e.g., approved, false_positive)
    </ResponseField>
    <ResponseField name="reviewer" type="string">
      ID of the user who reviewed the violation
    </ResponseField>
    <ResponseField name="note" type="string">
      Optional review note
    </ResponseField>
    <ResponseField name="timestamp" type="string">
      ISO 8601 timestamp of the review
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="summary" type="object">
    Summary statistics for the report
    <ResponseField name="total_violations" type="number">
      Total number of violations
    </ResponseField>
    <ResponseField name="critical_severity" type="number">
      Number of critical severity violations
    </ResponseField>
    <ResponseField name="high_severity" type="number">
      Number of high severity violations
    </ResponseField>
    <ResponseField name="medium_severity" type="number">
      Number of medium severity violations
    </ResponseField>
  </ResponseField>
</ResponseField>

## Error Responses

<ResponseField name="error" type="string">
  Error code: `NOT_FOUND`, `UNAUTHORIZED`, or `INTERNAL_ERROR`
</ResponseField>

<ResponseField name="message" type="string">
  Human-readable error message
</ResponseField>
